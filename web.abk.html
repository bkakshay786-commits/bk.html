from pathlib import Path import zipfile import sys # Configure paths (creates folder next to this script) project_dir = Path.cwd() / "birthday_wishes_web" project_dir.mkdir(parents=True, exist_ok=True) html_file = project_dir / "index.html" html_content = """ Birthday Wishes

Happy Birthday!

Wishing you a day full of happiness and a year full of joy.

Celebrate!

May your birthday be as amazing as you are.

Special Day

Cheers to another year of wonderful adventures.

Enjoy!

May your day be filled with love, laughter, and cake!

Best Wishes

Sending you smiles, hugs, and lots of birthday cheer!

""" try: html_file.write_text(html_content, encoding="utf-8") except Exception as e: print(f"Failed to write HTML file: {e}", file=sys.stderr) raise zip_path = Path.cwd() / "birthday_wishes_web.zip" try: with zipfile.ZipFile(zip_path, "w", compression=zipfile.ZIP_DEFLATED) as zf: zf.write(html_file, arcname="index.html") except Exception as e: print(f"Failed to create ZIP: {e}", file=sys.stderr) raise print(f"Project written to: {project_dir}") print(f"ZIP archive created: {zip_path}")
